filled circle:

```GLSL
void main() {
    vec2 pixel = gl_FragCoord.xy / u_resolution.xy;

    float value = 0.0;
    if (distance(pixel, vec2(0.5, 0.5)) < 0.37) {
        value = 1.0;
    }
    gl_FragColor = vec4(vec3(value), 1.0);
}
```





stripes:

```GLSL
void main() {
		vec2 pixel = gl_FragCoord.xy / u_resolution.xy;
    // pixel is normalize... x:(0, 1), y:(0, 1) 
    
    // rotate about center...
    pixel -= vec2(0.5);
    pixel = rotate(pixel, u_time * 0.1);
    pixel += vec2(0.5);
    
    // gl_FragColor = vec4(vec3(pow(sin(u_time), 2.0)), 1.0);
    float value = floor(2.0 * fract(pixel.x * 10.0));
    gl_FragColor = vec4(vec3(value), 1.0);
}
    
```



```GLSL
void main() {
  vec2 fc = gl_FragCoord.xy;
  //fc.x += 1.0;
  vec2 position = fc / u_resolution;
  position.x *= u_resolution.x / u_resolution.y;
  position *= 10.0;
  position.x *= -1.0;
  position = floor(position);
  float pattern = mod(position.x + mod(position.y, 2.0), 2.0);
  gl_FragColor = vec4(vec3(pattern), 1.0);
}
```





```GLSL
void main() {
  vec2 p = gl_FragCoord.xy / u_resolution;
  p += vec2(sin(u_time * 0.1), cos(u_time * 0.1)) * sin(u_time * 0.7);
  p = rotate(p, u_time * 0.15);
  p *= -1.0 + length(p) + 2.0 * sin(u_time * 0.5);
  vec2 f = mod(p * 35.0, 2.0) - vec2(1);
  //float x = 1.0 - pow(length(f), 3.0);
  float x = length(f) > 0.8 ? 0.0 : 1.0;
  gl_FragColor = vec4(vec3(x), 1.0);
}
```
